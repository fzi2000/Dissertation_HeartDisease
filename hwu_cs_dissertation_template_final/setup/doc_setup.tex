% set custom list of equations
\newcommand{\listequationsname}{List of Equations}
\newlistof{myequations}{equ}{\listequationsname}
\newcommand{\myequations}[1]{%
    \refstepcounter{myequations}
    {\centering \textbf{Equation \theequation}: #1 \par}
    \addcontentsline{equ}{myequations}
	{\hspace{1.5em}\protect\numberline{\theequation}#1}\par
}
\setlength{\cftmyequationsnumwidth}{2.5em}% Width of equation number in List of Equations

% set custom list of algorithms
\newcommand{\listalgorithmsname}{List of Algorithms}
\newlistof{myalgorithms}{alg}{\listalgorithmsname}
\newcommand{\myalgorithms}[1]{%
    \refstepcounter{myalgorithms}
    % \centering\textbf{Algorithm \thealgorithm}: #1
    \addcontentsline{alg}{myalgorithms}
	{\hspace{1.5em}\protect\numberline{\thealgorithm}#1}\par
}
\setlength{\cftmyalgorithmsnumwidth}{2.5em}% Width of algorithm number in List of Algorithms
